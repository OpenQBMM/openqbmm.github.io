102 void Foam::bubbleBreakupKernels::LuoSvendsen::update()
103 {
104  const phaseModel& phase(fluid_.phase1());
105  epsilonf_ = phase.turbulence().epsilon();
106  epsilonf_.max(SMALL);
107 
108  de_ = pow(pow3(phase.turbulence().nut())/epsilonf_, 0.25);
109 }

112 Foam::tmp<Foam::volScalarField>
113 Foam::bubbleBreakupKernels::LuoSvendsen::Kb(const label nodei) const
114 {
115  const volScalarField& d = fluid_.phase1().ds(nodei);
116  const volScalarField& rho2 = fluid_.phase2().rho();
117  const dimensionedScalar& sigma = fluid_.sigma();
118  volScalarField xi(de_/d + SMALL);
119 
120  tmp<volScalarField> breakupSource =
121  0.923*fluid_.phase2()*cbrt(epsilonf_*d)
122  *sqr(1.0 + xi)/(sqr(d)*pow(xi, 20.0))
123  *exp
124  (
125  12.0*Cf_*sigma
126  /(2.045*rho2*pow(xi, 11.0/3.0)*pow(d, 5.0/3.0)*pow(epsilonf_, 2.0/3.0))
127  );
128  breakupSource.ref().dimensions().reset(inv(dimTime));
129  return breakupSource;
130 }

134 (

138 {

140  scalar rho2(fluid_.phase2().rho()[celli]);
141  scalar sigma(fluid_.sigma().value());
142  scalar xi = max(de_[celli]/d, 20.0);
143 
144 
145  return
146  0.923*fluid_.phase2()[celli]*cbrt(epsilonf_[celli]*d)
147  *sqr(1.0 + xi)/(sqr(d)*pow(xi, 11.0/3.0))
148  *exp
149  (
150  -12.0*Cf_.value()*sigma
151  /(
