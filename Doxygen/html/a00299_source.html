126 void Foam::lognormalEQMOM::recurrenceRelation
127 (
128  scalarList& a,
129  scalarList& b,
130  scalar primaryAbscissa,
131  scalar sigma
132 )
133 {
134  scalar eta = exp(sqr(sigma)/2.0);
135  scalar sqEta = sqr(eta);
136 
137  a[0] = eta;
138 
139  for (label ai = 1; ai < a.size(); ai++)
140  {
141  a[ai] = ((sqEta + 1)*pow(sqEta, scalar(ai)) - 1.0)
142  *pow(eta, 2.0*scalar(ai) - 1.0);
143  }
144 
145  b[0] = 0.0;
146 
147  for (label bi = 1; bi < b.size(); bi++)
148  {
149  b[bi] = pow(eta, 6.0*scalar(bi) - 4.0)*(pow(sqEta, scalar(bi)) - 1.0);
150  }
151 }

153 Foam::scalar Foam::lognormalEQMOM::sigmaMax(univariateMomentSet& moments)
154 {
155  label nRealizableMoments = moments.nRealizableMoments();
156 
157  scalar sigmaZeta1 =
158  sqrt(2.0*log(sqrt(moments[0]*moments[2]/(sqr(moments[1])))));
159 
160  if (nRealizableMoments > 3)
161  {
162  scalar sigmaZeta2 =
163  sqrt(2.0*log(sqrt(moments[1]*moments[3]/(sqr(moments[2])))));
164 
165  return min(sigmaZeta1, sigmaZeta2);
166  }
167 
168  return sigmaZeta1;
169 }

172 {
173  tmp<scalarField> tmpY
174  (

177 
178  scalarField& y = tmpY.ref();
179
