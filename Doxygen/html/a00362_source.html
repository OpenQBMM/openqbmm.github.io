62 void Foam::PDFTransportModels::univariatePDFTransportModel::solve()
63 {
64  momentAdvection_().update();
65 
66  // List of moment transport equations
67  PtrList<fvScalarMatrix> momentEqns(quadrature_.nMoments());
68 
69  // Solve moment transport equations
70  forAll(quadrature_.moments(), momenti)
71  {
72  volScalarMoment& m = quadrature_.moments()[momenti];
73 
74  momentEqns.set
75  (
76  momenti,
77  new fvScalarMatrix
78  (
79  fvm::ddt(m)
80  + momentAdvection_().divMoments()[momenti]
81  ==
82  implicitMomentSource(m)
83  )
84  );
85  }
86  forAll (momentEqns, mEqni)
87  {
88  momentEqns[mEqni].relax();
89  momentEqns[mEqni].solve();
90  }
91  quadrature_.updateQuadrature();
92 
93  if (solveMomentSources())
94  {
95  this->explicitMomentSource();
96  }
97 }
