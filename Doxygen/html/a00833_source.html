59 void Foam::fluxFunctions::Roe::updateFluxes
60 (
61  surfaceScalarField& massFlux,
62  surfaceVectorField& momentumFlux,
63  surfaceScalarField& energyFlux,
64  const volScalarField& rho,
65  const volVectorField& U,
66  const volScalarField& H,
67  const volScalarField& p,
68  const volScalarField& a
69 )
70 {
71  surfaceScalarField rhoOwn
72  (
73  fvc::interpolate(rho, own_, interpScheme(rho.name()))
74  );
75  surfaceScalarField rhoNei
76  (
77  fvc::interpolate(rho, nei_, interpScheme(rho.name()))
78  );
79 
80  surfaceVectorField UOwn(fvc::interpolate(U, own_, interpScheme(U.name())));
81  surfaceVectorField UNei(fvc::interpolate(U, nei_, interpScheme(U.name())));
82 
83  surfaceScalarField HOwn(fvc::interpolate(H, own_, interpScheme(H.name())));
84  surfaceScalarField HNei(fvc::interpolate(H, nei_, interpScheme(H.name())));
85 
86  surfaceScalarField pOwn(fvc::interpolate(p, own_, interpScheme(p.name())));
87  surfaceScalarField pNei(fvc::interpolate(p, nei_, interpScheme(p.name())));
88 
89  surfaceScalarField aOwn
90  (
91  fvc::interpolate(a, own_, interpScheme(a.name()))
92  );
93  surfaceScalarField aNei
94  (
95  fvc::interpolate(a, nei_, interpScheme(a.name()))
96  );
97 
98  surfaceVectorField normal(mesh_.Sf()/mesh_.magSf());
99 
100  surfaceScalarField wOwn(sqrt(rhoOwn)/(sqrt(rhoOwn) + sqrt(rhoNei)));
101  surfaceScalarField wNei(1.0 - wOwn);
102 
103  surfaceScalarField rhoTilde
104  (
105  "rhoTilde",
106  sqrt
107  (
108  max
109  (
110  rhoOwn*rhoNei,
111  dimensionedScalar("SMALL", sqr(dimDensity), SMALL)
112  )
113  )
114  );
115 
116  surfaceVectorField UTilde
117  (
118  "UTilde",
119  UOwn*wOwn + UNei*wNei
120  );
121  surfaceScalarField UvTilde(UTilde & normal);
122  surfaceScalarField HTilde
123  (
124  "HTilde",
125  HOwn*wOwn + HNei*wNei
126  );
127  surfaceScalarField aTilde
128  (
129  "aTilde",
130  aOwn*wOwn + aNei*wNei
131  );
132 
133  surfaceScalarField deltaRho(rhoNei - rhoOwn);
134  surfaceVectorField deltaU(UNei - UOwn);
135  surfaceScalarField deltaUv(deltaU & normal);
136  surfaceScalarField deltaP(pNei - pOwn);
137 
138  surfaceScalarField lambda1(mag(UvTilde));
139  surfaceScalarField lambda2(mag(UvTilde + aTilde));
140  surfaceScalarField lambda3(mag(UvTilde - aTilde));
141 
142  surfaceScalarField alpha1
143  (
144  deltaRho - deltaP/sqr(aTilde)
145  );
146  surfaceScalarField alpha2
147  (
148  (deltaP + rhoTilde*aTilde*deltaUv)/(2.0*sqr(aTilde))
149  );
150  surfaceScalarField alpha3
151  (
152  (deltaP - rhoTilde*aTilde*deltaUv)/(2.0*sqr(aTilde))
153  );
154 
155  // U Row
156  surfaceVectorField K21(UTilde);
157  surfaceVectorField K224((UTilde + aTilde*normal));
158  surfaceVectorField K25((UTilde - aTilde*normal));
159 
160  // E row
161  surfaceScalarField K31(0.5*magSqr(UTilde));
162  surfaceScalarField K324((HTilde + aTilde*UvTilde));
163  surfaceScalarField K35((HTilde - aTilde*UvTilde));
164 
165  // Compute owner and neighbour fluxes
166  surfaceScalarField UvOwn(UOwn & normal);
167  surfaceScalarField UvNei(UNei & normal);
168 
169 // forAll(lambda1, facei)
170 // {
171 // scalar eps = 0.1*aTilde[facei]; //adjustable parameter
172 //
173 // if
174 // (
175 // lambda1[facei] < eps
176 // || lambda2[facei] < eps
177 // || lambda3[facei] < eps
178 // )
179 // {
180 // lambda1[facei] = (sqr(lambda1[facei]) + sqr(eps))/(2.0*eps);
181 // lambda2[facei] = (sqr(lambda2[facei]) + sqr(eps))/(2.0*eps);
182 // lambda3[facei] = (sqr(lambda3[facei]) + sqr(eps))/(2.0*eps);
183 // }
184 //
185 // // First eigenvalue: U - c
186 // eps =
187 // 2.0*max(0,(UvNei[facei] - aNei[facei])
188 // - (UvOwn[facei] - aOwn[facei]));
189 // if (lambda1[facei] < eps)
190 // {
191 // lambda1[facei] = (sqr(lambda1[facei]) + sqr(eps))/(2.0*eps);
192 // }
193 //
194 // // Second eigenvalue: U
195 // eps = 2*max(0, UvNei[facei] - UvOwn[facei]);
196 // if (lambda2[facei] < eps)
197 // {
198 // lambda2[facei] = (sqr(lambda2[facei]) + sqr(eps))/(2.0*eps);
199 // }
200 //
201 // // Third eigenvalue: U + c
202 // eps =
203 // 2.0*max(0,(UvNei[facei] + aNei[facei])
204 // - (UvOwn[facei] + aOwn[facei]));
205 // if (lambda3[facei] < eps)
206 // {
207 // lambda3[facei] = (sqr(lambda3[facei]) + sqr(eps))/(2.0*eps);
208 // }
209 // }

211  surfaceScalarField massFluxOwn(rhoOwn*UvOwn);
212  surfaceScalarField massFluxNei(rhoNei*UvNei);
213 
214  surfaceVectorField momentumFluxOwn
215  (
216  UOwn*massFluxOwn + pOwn*normal
217  );
218  surfaceVectorField momentumFluxNei
219  (
220  UNei*massFluxNei + pNei*normal
221  );
222 
223  surfaceScalarField energyFluxOwn(HOwn*massFluxOwn);
224  surfaceScalarField energyFluxNei(HNei*massFluxNei);
225 
226  // Compute fluxes
227  massFlux =
228  0.5*mesh_.magSf()
229  *(
230  massFluxOwn + massFluxNei
231  - (
232  mag(lambda1)*alpha1
233  + mag(lambda2)*alpha2
234  + mag(lambda3)*alpha3
235  )
236  );
237 
238  momentumFlux =
239  0.5*mesh_.magSf()
240  *(
241  momentumFluxOwn + momentumFluxNei
242  - (
243  mag(lambda1)*alpha1*K21
244  + mag(lambda2)*alpha2*K224
245  + mag(lambda3)*alpha3*K25
246  )
247  );
248 
249  energyFlux =
250  0.5*mesh_.magSf()
251  *(
252  energyFluxOwn + energyFluxNei
253  - (
254  mag(lambda1)*alpha1*K31
255  + mag(lambda2)*alpha2*K324
256  + mag(lambda3)*alpha3*K35
257  )
258  );
259 }
