130 void Foam::fv::phaseIncompressibleMeanVelocityForce::correct(volVectorField& U)
131 {
132  const scalarField& rAU = rAPtr_().internalField();
133 
134  // Integrate flow variables over cell set
135  scalar magAlphaUAve = 0.0;
136  scalar alphaAve = 0.0;
137  scalar rAUave = 0.0;
138  const scalarField& cv = mesh_.V();
139  forAll(cells_, i)
140  {
141  label cellI = cells_[i];
142  scalar volCell = cv[cellI];
143  magAlphaUAve += alpha_[cellI]*(flowDir_ & U[cellI])*volCell;
144  alphaAve += alpha_[cellI]*volCell;
145  rAUave += rAU[cellI]*volCell;
146  }
147 
148  // Collect across all processors
149  reduce(magAlphaUAve, sumOp<scalar>());
150  reduce(alphaAve, sumOp<scalar>());
151  reduce(rAUave, sumOp<scalar>());
152 
153  // Volume averages
154  magAlphaUAve /= V_;
155  alphaAve /= V_;
156  rAUave /= V_;
157 
158  // Calculate the pressure gradient increment needed to adjust the average
159  // flow-rate to the desired value
160  dGradP_ = relaxation_*(alphaAve*magUbar_ - magAlphaUAve)/rAUave;
161 
162  // Apply correction to velocity field
163  forAll(cells_, i)
164  {
165  label cellI = cells_[i];
166  U[cellI] += flowDir_*rAU[cellI]*dGradP_;
167  }
168 
169  scalar gradP = gradP0_ + dGradP_;
170 
171  Info<< "Pressure gradient source: uncorrected Mean Velocity Magnitude = "
172  << magAlphaUAve/alphaAve
173  << ", pressure gradient = " << gradP << endl;
174 
175  writeProps(gradP);
176 }
