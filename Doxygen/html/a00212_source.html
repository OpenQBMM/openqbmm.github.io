94 void Foam::multivariateMomentInversions::monoKinetic::invert
95 (
96  const multivariateMomentSet& moments
97 )
98 {
99  reset();
100 
101  univariateMomentSet sizeMoments(nSizeMoments_, "RPlus", 0.0);
102  forAll(sizeMoments, mi)
103  {
104  sizeMoments[mi] = moments(mi);
105  }
106 
107  sizeInverter_->invert(sizeMoments);
108  const scalarList& sizeWeights(sizeInverter_->weights());
109  const scalarList& sizeAbscissae(sizeInverter_->abscissae());
110 
111  forAll(sizeWeights, nodei)
112  {
113  weights_[nodei] = sizeWeights[nodei];
114  abscissae_[nodei][0] = sizeAbscissae[nodei];
115  }
116  label nSizeNodes = sizeWeights.size();
117 
118  if (nSizeNodes > 0)
119  {
120  scalarDiagonalMatrix x(nSizeNodes, 0.0);
121  scalarSquareMatrix invR(nSizeNodes, 0.0);
122 
123  forAll(sizeWeights, nodei)
124  {
125  x[nodei] = max(sizeAbscissae[nodei], small);
126  invR[nodei][nodei] = 1.0/max(sizeWeights[nodei], 1e-10);
127  }
128  Vandermonde V(x);
129  scalarSquareMatrix invVR = invR*V.inv();
130 
131  // Compute conditional velocity moments and invert
132  for (label dimi = 0; dimi < nvelocityDimensions_; dimi++)
133  {
134  labelList pureMomentOrder(nDistributionDims_, 0);
135  pureMomentOrder[dimi + 1] = 1;
136 
137  scalarRectangularMatrix M(nSizeNodes, 1, 0);
138  for (label nodei = 0; nodei < nSizeNodes; nodei++)
139  {
140  pureMomentOrder[0] = nodei;
141  M(nodei, 0) = moments(pureMomentOrder);
142  }
143  scalarRectangularMatrix nu = invVR*M;
144 
145  forAll(sizeWeights, nodei)
146  {
147  if (sizeWeights[nodei] > 1e-10)
148  {
149  velocityAbscissae_[nodei][dimi] = nu(nodei, 0);
150  }
151  }
152  }
153  }
154 }
