52 void Foam::kineticTheoryModels::anisotropicGaussian::updateh2Fn()
53 {
54  this->g0_ = this->radialModel_->g0
55  (
56  this->phase_,
57  this->alphaMinFriction_,
58  this->alphaMax_
59  );
60  PsFric_ = this->frictionalStressModel_->frictionalPressure
61  (
62  this->phase_,
63  this->alphaMinFriction_,
64  this->alphaMax_
65  );
66 
67  h2Fn_ = h2Function_->h2
68  (
69  this->phase_,
70  this->Theta_,
71  this->g0_,
72  this->phase_.rho(),
73  this->phase_.d(),
74  PsFric_,
75  this->e_
76  );
77  h2Fn_.max(this->residualAlpha_);
78  h2Fn_.correctBoundaryConditions();
79 }

166 void Foam::kineticTheoryModels::anisotropicGaussian::solve
167 (
168  const volScalarField& beta,
169  const volScalarField& alpha,
170  const volTensorField& gradU,
171  const volSymmTensorField D
172 )
173 {
174  // Local references
175  const volScalarField& rho = this->phase_.rho();
176  const surfaceScalarField& alphaRhoPhi = this->phase_.alphaRhoPhi();
177  const volVectorField& U = this->phase_.U();
178  const volVectorField& Uc = phase_.fluid().otherPhase(phase_).U();
179 
180  dimensionedScalar ThetaSmall("ThetaSmall", Theta_.dimensions(), 1.0e-6);
181  dimensionedScalar ThetaSmallSqrt(sqrt(ThetaSmall));
182  const scalar sqrtPi = sqrt(constant::mathematical::pi);
183  volScalarField ThetaSqrt("sqrtTheta", sqrt(Theta_));
184 
185  const volScalarField& da = this->phase_.d();
186  volSymmTensorField Sp(D - (1.0/3.0)*tr(D)*I);
187 
188 
189  // Drag
190  volScalarField rTauc
191  (
192  "rTauc",
193  6.0*sqrt(this->Theta_)
194  *max
195  (
196  alpha,
197  this->phase_.residualAlpha()
198  )*this->g0_/(da*sqrtPi)
199  );
200 
201  // Particle viscosity
202  this->nu_ *= (1.0 + 8.0/5.0*this->eta_*alpha*this->g0_)*h2Fn_;
203  this->nu_ += 3.0/5.0*this->lambda_;
204 
205  // 'thermal' conductivity
206  this->kappa_ *= (1.0 + 12.0/5.0*this->eta_*alpha*this->g0_)*h2Fn_;
207  this->kappa_ += 3.0/2.0*this->lambda_*rho;
208 
209  fv::options& fvOptions(fv::options::New(this->phase_.fluid().mesh()));
210 // const PhaseCompressibleTurbulenceModel<phaseModel>&
211 // particleTurbulenceModel =
212 // U.db().lookupObject<PhaseCompressibleTurbulenceModel<phaseModel> >
213 // (
214 // IOobject::groupName
215 // (
216 // turbulenceModel::propertiesName,
217 // this->phase_.name()
218 // )
219 // );
220 
221  // Solve Sigma equation (2nd order moments)
222  {
223  volSymmTensorField S2flux
224  (
225  "S2flux",
226  2.0*Sp
227  *(
228  this->granularPressureModel_->granularPressureCoeff
229  (
230  alpha,
231  this->g0_,
232  rho,
233  this->e_
234  )*this->Theta_
235  - rho*alpha*this->lambda_*tr(D)
236  )
237  - 2.0*rho*alpha*this->nu_
238  *(
239  twoSymm(Sp & gradU)
240  - (2.0/3.0)*(Sp && gradU)*I
241  )
242  );
243 
244  fvSymmTensorMatrix SigmaEqn
245  (
246  fvm::ddt(alpha, rho, Sigma_)
247  - fvc::ddt(alpha, rho, Sigma_)
248  + fvm::div(alphaRhoPhi, Sigma_)
249  - fvc::Sp
250  (
251  fvc::ddt(alpha, rho)
252  + fvc::div(alphaRhoPhi),
253  Sigma_
254  )
255  - fvm::laplacian
256  (
257  2.0/3.0*this->kappa_,
258 // + rho*particleTurbulenceModel.nut()/alphaSigma_,
259  Sigma_,
260  "laplacian(kappa,Sigma)"
261  )
262  ==
263  S2flux
264  - fvm::Sp
265  (
266  alpha
267  *(
268  2.0*beta + (3.0 - this->e_)*(1.0 + this->e_)/2.0*rTauc*rho
269  ),
270  Sigma_
271  )
272  + fvOptions(alpha, rho, Sigma_)
273 // + alpha*rho*particleTurbulenceModel.epsilon()
274  );
275 
276  SigmaEqn.relax();
277  fvOptions.constrain(SigmaEqn);
278  SigmaEqn.solve();
279  fvOptions.correct(Sigma_);
280  }
281 
282  // Solve granular temperature transport
283  // Stress tensor, Definitions, Table 3.1, p. 43
284  volSymmTensorField tau
285  (
286  rho*(2.0*nu_*D + (lambda_ - (2.0/3.0)*nu_)*tr(D)*I)
287  );
288 
289  // Dissipation (Eq. 3.24, p.50)
290  volScalarField gammaCoeff
291  (
292  "gammaCoeff",
293  12.0*(1.0 - sqr(e_))
294  *max(sqr(alpha), residualAlpha_)
295  *rho*g0_*(1.0/da)*ThetaSqrt/sqrtPi
296  );
297 
298  // Eq. 3.25, p. 50 Js = J1 - J2
299  volScalarField J1("J1", 3.0*beta);
300  volScalarField J2
301  (
302  "J2",
303  0.25*sqr(beta)*da*magSqr(U - Uc)
304  /(
305  max(alpha, residualAlpha_)*rho
306  *sqrtPi*(ThetaSqrt + ThetaSmallSqrt)
307  )
308  );
309 
310  // particle pressure - coefficient in front of Theta (Eq. 3.22, p. 45)
311  volScalarField PsCoeff
312  (
313  granularPressureModel_->granularPressureCoeff
314  (
315  alpha,
316  g0_,
317  rho,
318  e_
319  )
320  );
321 
322  // Construct the granular temperature equation (Eq. 3.20, p. 44)
323  // NB. note that there are two typos in Eq. 3.20:
324  // Ps should be without grad
325  // the laplacian has the wrong sign
326  fvScalarMatrix ThetaEqn
327  (
328  1.5*
329  (
330  fvm::ddt(alpha, rho, Theta_)
331  - fvc::ddt(alpha, rho, Theta_)
332  + fvm::div(alphaRhoPhi, Theta_)
333  - fvc::Sp(fvc::ddt(alpha, rho) + fvc::div(alphaRhoPhi), Theta_)
334  )
335  - fvm::laplacian
336  (
337  kappa_,
338 // + rho*particleTurbulenceModel.nut()/alphaTheta_,
339  Theta_,
340  "laplacian(kappa,Theta)"
341  )
342  ==
343  - fvm::SuSp((PsCoeff*I) && gradU, Theta_)
344  + (tau && gradU)
345  + fvm::Sp(-gammaCoeff, Theta_)
346  + fvm::Sp(-J1, Theta_)
347  + fvm::Sp(J2/(Theta_ + ThetaSmall), Theta_)
348 // + alpha*rho*particleTurbulenceModel.epsilon()
349 
350  + fvOptions(alpha, rho, Theta_)
351  );
352 
353  ThetaEqn.relax();
354  fvOptions.constrain(ThetaEqn);
355  ThetaEqn.solve();
356  fvOptions.correct(Theta_);
357 
358  Theta_.max(0);
359  Theta_.min(100);
360 
361  PsFric_ =
362  frictionalStressModel_->frictionalPressurePrime
363  (
364  phase_,
365  alphaMinFriction_,
366  alphaMax_
367  );
368 
369  if (debug)
370  {
371  Info<< " max(Sigma) = " << max(mag(Sigma_)).value()
372  << " max(Theta) = " << max(Theta_).value() << endl;
373  }
374 }
