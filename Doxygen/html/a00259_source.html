310 void Foam::betaEQMOM::recurrenceRelation
311 (
312  scalarList& a,
313  scalarList& b,
314  scalar primaryAbscissa,
315  scalar sigma
316 )
317 {
318  scalar alpha = (1.0 - primaryAbscissa)/sigma - 1.0;
319  scalar beta = primaryAbscissa/sigma - 1.0;
320  scalar nab = 0.0;
321 
322  a[0] = (beta - alpha)/(alpha + beta + 2.0);
323 
324  b[0] = exp((alpha + beta + 1.0)*log(2.0) + lgamma(alpha + 1.0)
325  + lgamma(beta + 1.0) - lgamma(alpha + beta + 2.0));
326 
327  a[1] = (sqr(beta) - sqr(alpha))/((alpha + beta + 2.0)*(alpha + beta + 4.0));
328 
329  b[1] = (4.0*(alpha + 1.0)*(beta + 1.0))
330  /(sqr(alpha + beta + 2.0)*(alpha + beta + 3.0));
331 
332  for (label i = 2; i < a.size(); i++)
333  {
334  nab = 2.0*scalar(i) + alpha + beta;
335 
336  a[i] = (sqr(beta) - sqr(alpha))/(nab*(nab + 2.0));
337 
338  b[i] = (4.0*(scalar(i) + alpha)*(scalar(i) + beta)*scalar(i)
339  *(scalar(i)+ alpha + beta))/(sqr(nab)*(nab + 1.0)*(nab - 1.0));
340  }
341 }
