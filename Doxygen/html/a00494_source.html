54 void
55 Foam::populationBalanceSubModels::collisionKernels::BoltzmannCollision::updateI
56 (
57  const label celli,
58  const label node1,
59  const label node2,
60  const scalar omega
61 )
62 {
63  const vector& u1 = quadrature_.nodes()[node1].velocityAbscissae()[celli];
64  const vector& u2 = quadrature_.nodes()[node2].velocityAbscissae()[celli];
65 
66  // Store powers of velocities to reduce evaluations
67  vector g = u1 - u2;
68  scalarList omegaPow(6, omega);
69  vectorList gPow(6, g);
70  scalarList gMagPow(6, mag(g));
71  vectorList vPow(6, u1);
72 
73  for (label powi = 2; powi < gPow.size(); powi++)
74  {
75  omegaPow[powi] = pow(omega, powi);
76  gMagPow[powi] = pow(mag(g), powi);
77  for (label cmpt = 0; cmpt < nDimensions_; cmpt++)
78  {
79  gPow[powi][cmpt] = pow(g[cmpt], powi);
80  vPow[powi][cmpt] = pow(u1[cmpt], powi);
81  }
82  }
83 
84  // Update coefficients for zero order terms
85  forAllIter(List<momentFunction>, coefficientFunctions_, iter)
86  {
87  (*iter)(Is_, omegaPow, gPow, gMagPow, vPow);
88  }
89 
90  // Update first order (Enskog) terms if used
91  if (Enskog_)
92  {
93  forAllIter(List<momentFunction>, enskogFunctions_[0], iter)
94  {
95  (*iter)(I1s_[0], omegaPow, gPow, gMagPow, vPow);
96  }
97 
98  if (nDimensions_ > 1)
99  {
100  forAllIter(List<momentFunction>, enskogFunctions_[1], iter)
101  {
102  (*iter)(I1s_[1], omegaPow, gPow, gMagPow, vPow);
103  }
104  }
105  if (nDimensions_ > 2)
106  {
107  forAllIter(List<momentFunction>, enskogFunctions_[2], iter)
108  {
109  (*iter)(I1s_[2], omegaPow, gPow, gMagPow, vPow);
110  }
111  }
112  }
113 }

327 void Foam::populationBalanceSubModels::collisionKernels::BoltzmannCollision
328 ::updateCells
329 (
330  const label celli
331 )
332 {
333  forAll(Cs_, momenti)
334  {
335  Cs_[momenti][celli] = 0.0;
336  }
337  if (Enskog_)
338  {
339  forAll(Gs_, momenti)
340  {
341  Gs_[momenti][celli] = Zero;
342  }
343  }
344 
345  scalar alpha = quadrature_.moments()(0)[celli];
346  scalar alphac = 1.0 - alpha;
347 
348  // Monodisperse case
349  if (sizeIndex_ == -1)
350  {
351  scalar c = min(alpha/0.63, 0.999);
352  scalar g0 = (2.0 - c)/(2.0*pow3(1.0 - c));
353  forAll(quadrature_.nodes(), nodei)
354  {
355  const volVelocityNode& node1 = quadrature_.nodes()[nodei];
356  forAll(quadrature_.nodes(), nodej)
357  {
358  const volVelocityNode& node2 = quadrature_.nodes()[nodej];
359 
360  updateI(celli, nodei, nodej, omega_);
361 
362  forAll(Cs_, momenti)
363  {
364  const labelList& momentOrder = momentOrders_[momenti];
365  labelList vMomentOrder(velocityIndexes_.size(), 0);
366  forAll(velocityIndexes_, cmpt)
367  {
368  vMomentOrder[cmpt] = momentOrder[velocityIndexes_[cmpt]];
369  }
370 
371  //- Zero order source term
372  scalar cSource =
373  6.0*g0/dp_()[celli]
374  *node1.primaryWeight()[celli]
375  *node2.primaryWeight()[celli]
376  *mag
377  (
378  node1.velocityAbscissae()[celli]
379  - node2.velocityAbscissae()[celli]
380  )
381  *Is_(vMomentOrder);
382 
383  //- Enskog term
384  if (Enskog_)
385  {
386  scalar eSource = 0.0;
387  forAll(velocityIndexes_, m)
388  {
389  scalar I1m = I1s_[m](vMomentOrder);
390  eSource +=
391  I1m
392  *(
393  node2.primaryWeight()[celli]
394  *gradWs_[nodei][celli][m]
395  - node1.primaryWeight()[celli]
396  *gradWs_[nodej][celli][m]
397  );
398  Gs_[momenti][celli][m] +=
399  3.0*g0
400  *I1m
401  *node1.primaryWeight()[celli]
402  *node2.primaryWeight()[celli];
403  }
404  cSource += 3.0*g0*eSource;
405  }
406 
407  // Integrate over non-velocity components
408  forAll(scalarIndexes_, cmpt)
409  {
410  scalar absCmpt =
411  pow
412  (
413  node1.primaryAbscissae()[scalarIndexes_[cmpt]][celli],
414  momentOrder[scalarIndexes_[cmpt]]
415  );
416  cSource *= absCmpt;
417 
418  if (Enskog_)
419  {
420  Gs_[momenti][celli] *= absCmpt;
421  }
422  }
423  Cs_[momenti][celli] += cSource;
424  }
425  }
426  }
427 
428  return;
429  }
430 
431  // Polydisperse case
432  scalar pi = Foam::constant::mathematical::pi;
433  forAll(quadrature_.nodes(), nodei)
434  {
435  const label sizei = nodeIndexes_[nodei][sizeIndex_];
436  const volVelocityNode& node1 = quadrature_.nodes()[nodei];
437  scalar d1 = d(sizei, celli);
438  scalar V1 = pi/6.0*pow3(d1);
439  scalar mass1 = V1*rhos_[sizei];
440  scalar n1 = node1.primaryWeight()[celli]/V1;
441 
442  forAll(quadrature_.nodes(), nodej)
443  {
444  const label sizej = nodeIndexes_[nodej][sizeIndex_];
445  const volVelocityNode& node2 = quadrature_.nodes()[nodej];
446  scalar d2 = d(sizej, celli);
447  scalar V2 = pi/6.0*pow3(d2);
448  scalar mass2 = V2*rhos_[sizej];
449  scalar n2 = node2.primaryWeight()[celli]/V2;
450 
451  scalar d12 = (d1 + d2)*0.5;
452  scalar XiSqr = sqr(d12/d2);
453  scalar omega = mass2*(1.0 + e_)/(mass1 + mass2);
454 
455  scalar xi = pi*(n1*sqr(d1) + n2*sqr(d2))/6.0;
456  scalar g012 =
457  1.0/alphac
458  + 1.5*xi*d1*d2/(sqr(alphac)*(d12))
459  + 0.5*sqr(xi)/pow3(alphac)*sqr(d1*d2/d12);
460 
461  if (omega > small)
462  {
463  updateI(celli, nodei, nodej, omega);
464 
465  forAll(Cs_, momenti)
466  {
467  const labelList& momentOrder = momentOrders_[momenti];
468  labelList vMomentOrder(velocityIndexes_.size(), 0);
469  forAll(velocityIndexes_, cmpt)
470  {
471  vMomentOrder[cmpt] = momentOrder[velocityIndexes_[cmpt]];
472  }
473 
474  //- Zero order source term
475  scalar cSource =
476  6.0*XiSqr*g012/d2
477  *node1.primaryWeight()[celli]
478  *node2.primaryWeight()[celli]
479  *mag
480  (
481  node1.velocityAbscissae()[celli]
482  - node2.velocityAbscissae()[celli]
483  )
484  *Is_(vMomentOrder);
485 
486  //- Enskog term
487  if (Enskog_)
488  {
489  scalar enskogCoeff = 3.0*XiSqr*g012*d1/d2;
490  scalar eSource = 0.0;
491  forAll(velocityIndexes_, m)
492  {
493  scalar I1m = I1s_[m](vMomentOrder);
494  eSource +=
495  I1m
496  *(
497  d1*node2.primaryWeight()[celli]
498  *gradWs_[nodei][celli][m]
499  - d2*node1.primaryWeight()[celli]
500  *gradWs_[nodej][celli][m]
501  );
502  Gs_[momenti][celli][m] =
503  enskogCoeff
504  *I1m
505  *node1.primaryWeight()[celli]
506  *node2.primaryWeight()[celli];
507  }
508  cSource += enskogCoeff*eSource;
509  }
510 
511  // Integrate over non-velocity abscissae
512  forAll(scalarIndexes_, cmpt)
513  {
514  scalar absCmpt =
515  pow
516  (
517  node1.primaryAbscissae()[scalarIndexes_[cmpt]][celli],
518  momentOrder[scalarIndexes_[cmpt]]
519  );
520  cSource *= absCmpt;
521 
522  if (Enskog_)
523  {
524  Gs_[momenti][celli] *= absCmpt;
525  }
526  }
527  Cs_[momenti][celli] += cSource;
528  }
529  }
530  }
531  }
532 }
