36 Foam::momentFieldSet<momentType, nodeType>::momentFieldSet
37 (
38  const word& distributionName,
39  const dictionary& dict,
40  const fvMesh& mesh,
41  const autoPtr<mappedPtrList<nodeType>>& nodes,
42  const word& support
43 )
44 :
45  mappedPtrList<momentType>
46  (
47  dict.lookup("moments"),
48  typename momentType::iNew(distributionName, mesh, nodes)

50  name_(IOobject::groupName("moments", distributionName)),
51  nodes_(nodes),
52  nDimensions_((*this)[0].nDimensions()),
53  nMoments_((*this).size()),
