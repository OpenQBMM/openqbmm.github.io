85 Foam::tmp<Foam::volScalarField> Foam::dragModels::segregated::K
86 (
87  const label nodei,
88  const label nodej
89 ) const
90 {
91  const fvMesh& mesh(pair_.phase1().mesh());
92 
93  volScalarField alpha1(pair_.phase1().volumeFraction());
94  volScalarField alpha2(pair_.phase2().volumeFraction());
95 
96  tmp<volScalarField> trho1(pair_.phase1().rho());
97  tmp<volScalarField> trho2(pair_.phase2().rho());
98 
99  tmp<volScalarField> tnu1(pair_.phase1().nu());
100  tmp<volScalarField> tnu2(pair_.phase2().nu());
101 
102  const volScalarField& rho1(trho1());
103  const volScalarField& rho2(trho2());
104 
105  const volScalarField& nu1(tnu1());
106  const volScalarField& nu2(tnu2());
107 
108  volScalarField L
109  (
110  IOobject
111  (
112  "L",
113  mesh.time().timeName(),
114  mesh
115  ),
116  mesh,
117  dimensionedScalar("L", dimLength, 0),
118  zeroGradientFvPatchField<scalar>::typeName
119  );
120 
121  L.primitiveFieldRef() = cbrt(mesh.V());
122  L.correctBoundaryConditions();
123 
124  volScalarField I
125  (
126  alpha1
127  /max
128  (
129  alpha1 + alpha2,
130  pair_.phase1().residualAlpha() + pair_.phase2().residualAlpha()
131  )
132  );
133 
134  if (pair_.phase2().nNodes() > 1)
135  {
136  // Scale so not counted nNodes times
137  I *=
138  alpha2
139  /max
140  (
141  pair_.phase2(),
142  pair_.phase2().residualAlpha()
143  );
144  }
145 
146  volScalarField magGradI
147  (
148  max
149  (
150  mag(fvc::grad(I)),
151  (pair_.phase1().residualAlpha() + pair_.phase2().residualAlpha())/L
152  )
153  );
154 
155  volScalarField muI
156  (
157  rho1*nu1*rho2*nu2
158  /(rho1*nu1 + rho2*nu2)
159  );
160 
161  volScalarField muAlphaI
162  (
163  alpha1*rho1*nu1*alpha2*rho2*nu2
164  /(
165  max(alpha1, pair_.phase1().residualAlpha())*rho1*nu1
166  + max(alpha2, pair_.phase2().residualAlpha())*rho2*nu2
167  )
168  );
169 
170  volScalarField ReI
171  (
172  pair_.rho()
173  *pair_.magUr(nodei, nodej)
174  /(magGradI*muI)
175  );
176 
177  volScalarField lambda(m_*ReI + n_*muAlphaI/muI);
178 
179  return lambda*sqr(magGradI)*muI;
180 }
