52 Foam::populationBalanceSubModels::collisionKernels::esBGKCollision::covariance
53 (
54  const label celli,
55  const scalar& u,
56  const scalar& v,
57  const scalar& w
58 )
59 {
60  symmTensor sigma(Zero);
61 
62  const volVelocityMomentFieldSet& moments = quadrature_.moments();
63  scalar m0 = max(moments(0)[celli], SMALL);
64 
65  // Variances of velocities
66  scalar sigma1 = max(moments(2)[celli]/m0 - sqr(u), 0.0);
67  scalar sigma2 = 0.0;
68  scalar sigma3 = 0.0;
69  Theta_[celli] = sigma1;
70 
71  if (nDimensions_ > 1)
72  {
73  sigma2 = max(moments(0,2)[celli]/m0 - sqr(v), 0.0);
74  Theta_[celli] += sigma2;
75  }
76  if (nDimensions_ > 2)
77  {
78  sigma3 = max(moments(0,0,2)[celli]/m0 - sqr(w), 0.0);
79  Theta_[celli] += sigma3;
80  }
81  Theta_[celli] /= nDimensions_;
82 
83  sigma.xx() = a1_*Theta_[celli] + b1_*sigma1;
84 
85  if (nDimensions_ > 1)
86  {
87  sigma.yy() = a1_*Theta_[celli] + b1_*sigma2;
88  sigma.xy() = b1_*(moments(1,1)[celli]/m0 - u*v);
89  }
90 
91  if (nDimensions_ > 2)
92  {
93  sigma.zz() = a1_*Theta_[celli] + b1_*sigma3;
94  sigma.xz() = b1_*(moments(1,0,1)[celli]/m0 - u*w);
95  sigma.yz() = b1_*(moments(0,1,1)[celli]/m0 - v*w);
96  }
97 
98  return sigma;
99 }

192 Foam::scalar
193 Foam::populationBalanceSubModels::collisionKernels::esBGKCollision
194 ::explicitCollisionSource
195 (
196  const labelList& momentOrder,
197  const label celli
198 ) const
199 {
200  if (implicit_)
201  {
202  return 0.0;
203  }
204  if (nSizes_ > 0)
205  {
206  return Meq_(momentOrder)[celli];
207  }
208 
209  scalar c = quadrature_.moments()(0)[celli]/0.63;
210  scalar gs0 = (2.0 - c)/(2.0*pow3(1.0 - c)) + 1.1603*c;
211 
212  scalar tauC =
213  zeta_*sqrt(Foam::constant::mathematical::pi)*dp_()[celli]
214  /max
215  (
216  12.0*gs0*quadrature_.moments()[0][celli]*sqrt(Theta_[celli]),
217  1e-10
218  );
219 
220  return
221  (
222  Meq_(momentOrder)[celli]
223  - quadrature_.moments()(momentOrder)[celli]
224  )/tauC;
225 }
