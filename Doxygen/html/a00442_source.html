66 Foam::tmp<Foam::volScalarField> Foam::liftModels::LegendreMagnaudet::Cl
67 (
68  const label nodei,
69  const label nodej
70 ) const
71 {
72  volScalarField Re(max(pair_.Re(nodei, nodej), residualRe_));
73 
74  volScalarField Sr
75  (
76  sqr(pair_.dispersed().ds(nodei))
77  /(
78  Re
79  *pair_.continuous().nu()
80  )
81  *mag(fvc::grad(pair_.continuous().Us(nodej)))
82  );
83 
84  volScalarField ClLowSqr
85  (
86  sqr(6.0*2.255)
87  *sqr(Sr)
88  /(
89  pow4(constant::mathematical::pi)
90  *Re
91  *pow3(Sr + 0.2*Re)
92  )
93  );
94 
95  volScalarField ClHighSqr
96  (
97  sqr(0.5*(Re + 16.0)/(Re + 29.0))
98  );
99 
100  return sqrt(ClLowSqr + ClHighSqr);
101 }
