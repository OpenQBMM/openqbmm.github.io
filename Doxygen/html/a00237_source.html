91 void Foam::univariateMomentInversion::invert
92 (
93  univariateMomentSet& moments,
94  const scalar minKnownAbscissa,
95  const scalar maxKnownAbscissa
96 )
97 {
98  if (moments.isDegenerate())
99  {
100  nNodes_ = 1;
101  weights_.setSize(nNodes_);
102  abscissae_.setSize(nNodes_);
103  weights_[0] = moments[0];
104  abscissae_[0] = 0.0;
105 
106  return;
107  }
108 
109  if (moments[0] < SMALL)
110  {
111  nNodes_ = 0;
112 
113  weights_.setSize(nNodes_);
114  abscissae_.setSize(nNodes_);
115 
116  return;
117  }
118 
119  calcNQuadratureNodes(moments);
120 
121  if (nInvertibleMoments_ == 2)
122  {
123  weights_[0] = moments[0];
124  abscissae_[0] = moments[1]/moments[0];
125 
126  return;
127  }
128 
129  scalarSquareMatrix z(nNodes_, scalar(0));
130  JacobiMatrix(moments, z, minKnownAbscissa, maxKnownAbscissa);
131 
132  // Computing weights and abscissae
133  eigenSolver zEig(z, true);
134 
135  // Computing weights and abscissae
136  for (label i = 0; i < nNodes_; i++)
137  {
138  weights_[i] = moments[0]*sqr(zEig.eigenvectors()[0][i]);
139  abscissae_[i] = zEig.eigenvaluesRe()[i];
140  }
141 }
