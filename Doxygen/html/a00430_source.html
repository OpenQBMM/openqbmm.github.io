286 void Foam::kineticTheoryModel::update()
287 {
288  g0_ = radialModel_->g0
289  (
290  max(phase_, scalar(0)),
291  alphaMinFriction_,
292  phase_.alphaMax()
293  );
294 
295  // particle viscosity (Table 3.2, p.47)
296  nu_ = viscosityModel_->nu
297  (
298  phase_,
299  Theta_,
300  g0_,
301  phase_.rho(),
302  phase_.d(),
303  e_
304  );
305 
306  // Bulk viscosity p. 45 (Lun et al. 1984).
307  lambda_ =
308  (4.0/3.0)*sqr(phase_)*phase_.d()*g0_*(1.0 + e_)
309  *sqrt(Theta_/constant::mathematical::pi);
310 
311  // 'thermal' conductivity
312  kappa_ = conductivityModel_->kappa
313  (
314  phase_,
315  Theta_,
316  g0_,
317  phase_.rho(),
318  phase_.d(),
319  e_
320  );
321 
322  nuFric_ = frictionalStressModel_->nu
323  (
324  phase_,
325  alphaMinFriction_,
326  alphaMax_,
327  frictionalStressModel_->frictionalPressure
328  (
329  phase_,
330  alphaMinFriction_,
331  alphaMax_
332  )/phase_.rho(),
333  symm(fvc::grad(phase_.U()))
334  );
335 
336  // Limit viscosity and add frictional viscosity
337  nu_.min(maxNut_);
338  nuFric_ = min(nuFric_, maxNut_ - nu_);
339 
340  if (debug)
341  {
342  Info<< " max(nu) = " << max(nu_).value() << nl
343  << " max(nuFric) = " << max(nuFric_).value() << endl;
344  }
345 }
