237 void Foam::PDFTransportModels::populationBalanceModels::mixingPopulationBalance
238 ::calcEnvironmentMoments()
239 {
240  const volScalarMomentFieldSet& mXi
241  (
242  mixingModel_().quadrature().moments()
243  );
244 
245  const volScalarMoment& xiMean_ = mXi[1];
246  const volScalarMoment& xiMTwo_ = mXi[2];
247 
248  // Compute variance of the mixture fraction
249  const volScalarField xiVariance(xiMTwo_ - sqr(xiMean_));
250 
251  // Difference between abscissae of the mixture fraction
252  volScalarField xiDiff(xi1_ - xi2_);
253 
254  // Compute moments in the environment
255  forAll(xiDiff, celli)
256  {
257  // Null or very small variance of the mixture fraction
258  // Moments in the two environments are identical
259  if (xiDiff[celli] > minMixtureFractionVariance_)
260  {
261  forAll(mEnvOne_, mi)
262  {
263  if (p1_[celli] > minEnvironmentWeight_)
264  {
265  mEnvOne_[mi][celli]
266  =
267  (
268  meanMomentsVariance_[mi][celli]
269  - meanMoments_[mi][celli]*xi2_[celli]
270  )/(p1_[celli]*xiDiff[celli]);
271  }
272  else
273  {
274  mEnvOne_[mi][celli] = 0;
275  }
276 
277  if (p2_[celli] > minEnvironmentWeight_)
278  {
279  mEnvTwo_[mi][celli]
280  =
281  (
282  meanMoments_[mi][celli]*xi1_[celli]
283  - meanMomentsVariance_[mi][celli]
284  )/(p2_[celli]*xiDiff[celli]);
285  }
286  else
287  {
288  mEnvTwo_[mi][celli] = 0;
289  }
290  }
291  }
292  else
293  {
294  forAll(mEnvOne_, mi)
295  {
296  mEnvOne_[mi][celli] = meanMoments_[mi][celli];
297  mEnvTwo_[mi][celli] = meanMoments_[mi][celli];
298  }
299  }
300  }
301 
302  forAll(mEnvOne_, mi)
303  {
304  mEnvOne_[mi].correctBoundaryConditions();
305  mEnvTwo_[mi].correctBoundaryConditions();
306  }
307 }

310 void Foam::PDFTransportModels::populationBalanceModels::mixingPopulationBalance
311 ::calcMixedMoments()
312 {
313  forAll(meanMoments_, mi)
314  {
315  meanMoments_[mi] == p1_*mEnvOne_[mi] + p2_*mEnvTwo_[mi];
316 
317  meanMomentsVariance_[mi] ==
318  p1_*xi1_*mEnvOne_[mi] + p2_*xi2_*mEnvTwo_[mi];
319  }

400 void Foam::PDFTransportModels::populationBalanceModels::mixingPopulationBalance
401 ::solve()
402 {
403  // Solve mixingModel
404  Info << "Solving mixing model" << endl;
405 
406  mixingModel_().solve();
407 
408  Info << "Solving population balance" << endl;
409 
410  // Update PBE advection
411  meanMomentsAdvection_().update();
412  meanMomentsVarianceAdvection_().update();
413 
414  // List of moment transport equations for the mean moments
415  PtrList<fvScalarMatrix> meanMomentEqns
416  (
417  meanMomentsQuadrature_.nMoments()
418  );
419 
420  // List of moment transport equations for the variance of the mean moments
421  PtrList<fvScalarMatrix> meanMomentVarianceEqns
422  (
423  meanMomentsVarianceQuadrature_.nMoments()
424  );
425 
426  // Solve moment transport equations
427  forAll(meanMomentsQuadrature_.moments(), momenti)
428  {
429  volScalarMoment& meanM
430  (
431  meanMomentsQuadrature_.moments()[momenti]
432  );
433 
434  meanMomentEqns.set
435  (
436  momenti,
437  new fvScalarMatrix
438  (
439  fvm::ddt(meanM)
440  + meanMomentsAdvection_().divMoments()[momenti]
441  - diffusionModel_->momentDiff(meanM)
442  )
443  );
444 
445  volScalarMoment& varM
446  (
447  meanMomentsVarianceQuadrature_.moments()[momenti]
448  );
449 
450  meanMomentVarianceEqns.set
451  (
452  momenti,
453  new fvScalarMatrix
454  (
455  fvm::ddt(varM)
456  + meanMomentsVarianceAdvection_().divMoments()[momenti]
457  - diffusionModel_->momentDiff(varM)
458  ==
459  envMixingModel_->K
460  (
461  meanM,
462  varM,
463  meanXi_
464  )
465  )
466  );
467  }
468 
469  // Update moment environments to apply ODE solver
470  calcEnvironmentMoments();
471 
472  // Solve source terms
473  odeType::solve(envOneQuadrature_, 1);
474  odeType::solve(envTwoQuadrature_, 2);
475 
476  // Update mixed moments
477  calcMixedMoments();
478 
479  // Finish solving for moments
480  forAll (meanMomentEqns, mEqni)
481  {
482  const volScalarMoment& meanM
483  (
484  meanMomentsQuadrature_.moments()[mEqni]
485  );
486 
487  const volScalarMoment& varM
488  (
489  meanMomentsVarianceQuadrature_.moments()[mEqni]
490  );
491 
492  meanMomentEqns[mEqni] -= fvc::ddt(meanM);
493  meanMomentVarianceEqns[mEqni] -= fvc::ddt(varM);
494 
495  meanMomentVarianceEqns[mEqni].relax();
496  meanMomentVarianceEqns[mEqni].solve();
497 
498  meanMomentEqns[mEqni].relax();
499  meanMomentEqns[mEqni].solve();
500  }
501 
502  meanMomentsQuadrature_.updateQuadrature();
503  meanMomentsVarianceQuadrature_.updateQuadrature();
504 }
