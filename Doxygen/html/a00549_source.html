83 void Foam::kineticTheoryModels::nonEquilibrium::solve
84 (
85  const volScalarField& beta,
86  const volScalarField& alpha,
87  const volTensorField& gradU,
88  const volSymmTensorField D
89 )
90 {
91  // Local references
92  const volScalarField& rho = phase_.rho();
93  const surfaceScalarField& alphaRhoPhi = phase_.alphaRhoPhi();
94  const volVectorField& U = phase_.U();
95  const volVectorField& Uc = phase_.fluid().otherPhase(phase_).U();
96 
97  const scalar sqrtPi = sqrt(constant::mathematical::pi);
98  dimensionedScalar ThetaSmall("ThetaSmall", Theta_.dimensions(), 1.0e-6);
99  dimensionedScalar ThetaSmallSqrt(sqrt(ThetaSmall));
100 
101  const volScalarField& da = phase_.d();
102 
103  volScalarField ThetaSqrt("sqrtTheta", sqrt(Theta_));
104 
105  // Stress tensor, Definitions, Table 3.1, p. 43
106  volSymmTensorField tau
107  (
108  rho*(2.0*nu_*D + (lambda_ - (2.0/3.0)*nu_)*tr(D)*I)
109  );
110 
111  // Dissipation (Eq. 3.24, p.50)
112  volScalarField gammaCoeff
113  (
114  "gammaCoeff",
115  12.0*(1.0 - sqr(e_))
116  *max(sqr(alpha), residualAlpha_)
117  *rho*g0_*(1.0/da)*ThetaSqrt/sqrtPi
118  );
119 
120  // Eq. 3.25, p. 50 Js = J1 - J2
121  volScalarField J1("J1", 3.0*beta);
122  volScalarField J2
123  (
124  "J2",
125  0.25*sqr(beta)*da*magSqr(U - Uc)
126  /(
127  max(alpha, residualAlpha_)*rho
128  *sqrtPi*(ThetaSqrt + ThetaSmallSqrt)
129  )
130  );
131 
132  // particle pressure - coefficient in front of Theta (Eq. 3.22, p. 45)
133  volScalarField PsCoeff
134  (
135  granularPressureModel_->granularPressureCoeff
136  (
137  alpha,
138  g0_,
139  rho,
140  e_
141  )
142  );
143 
144  fv::options& fvOptions(fv::options::New(phase_.fluid().mesh()));
145  const PhaseCompressibleTurbulenceModel<phaseModel>&
146  particleTurbulenceModel =
147  U.db().lookupObject<PhaseCompressibleTurbulenceModel<phaseModel> >
148  (
149  IOobject::groupName
150  (
151  turbulenceModel::propertiesName,
152  phase_.name()
153  )
154  );
155 
156  // Construct the granular temperature equation (Eq. 3.20, p. 44)
157  // NB. note that there are two typos in Eq. 3.20:
158  // Ps should be without grad
159  // the laplacian has the wrong sign
160  fvScalarMatrix ThetaEqn
161  (
162  1.5*
163  (
164  fvm::ddt(alpha, rho, Theta_)
165  - fvc::ddt(alpha, rho, Theta_)
166  + fvm::div
167  (
168  this->h2f()*alphaRhoPhi,
169  Theta_,
170  "div(" + alphaRhoPhi.name() + "," + Theta_.name() + ")"
171  )
172  - fvc::Sp(fvc::ddt(alpha, rho) + fvc::div(alphaRhoPhi), Theta_)
173  )
174  - fvm::laplacian
175  (
176  kappa_,
177 // + rho*particleTurbulenceModel.nut()/alphaTheta_,
178  Theta_,
179  "laplacian(kappa,Theta)"
180  )
181  ==
182  - fvm::SuSp((PsCoeff*I) && gradU, Theta_)
183  + (tau && gradU)
184  + fvm::Sp(-gammaCoeff, Theta_)
185  + fvm::Sp(-J1, Theta_)
186  + fvm::Sp(J2/(Theta_ + ThetaSmall), Theta_)
187  + alpha*rho*particleTurbulenceModel.epsilon()
188 
189  + fvOptions(alpha, rho, Theta_)
190  );
191 
192  ThetaEqn.relax();
193  fvOptions.constrain(ThetaEqn);
194  ThetaEqn.solve();
195  fvOptions.correct(Theta_);
196 
197  Theta_.max(0);
198  Theta_.min(100);
199 
200  if (debug)
201  {
202  Info<< " max(Theta) = " << max(Theta_).value() << endl;
203  }
204 }
