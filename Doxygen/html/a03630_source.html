65 Foam::tmp<Foam::volScalarField> Foam::liftModels::Moraga::Cl
66 (
67  const label nodei,
68  const label nodej
69 ) const
70 {
71  volScalarField Re(pair_.Re(nodei, nodej));
72 
73  volScalarField sqrSr
74  (
75  sqr(pair_.dispersed().ds(nodei))
76  /pair_.continuous().nu()
77  *mag(fvc::grad(pair_.continuous().Us(nodej)))
78  );
79 
80  if
81  (
82  min(Re).value() < 1200.0
83  || max(Re).value() > 18800.0
84  || min(sqrSr).value() < 0.0016
85  || max(sqrSr).value() > 0.04
86  )
87  {
88  WarningInFunction
89  << "Re and/or Sr are out of the range of applicability of the "
90  << "Moraga model. Clamping to range bounds"
91  << endl;
92  }
93 
94  Re.min(1200.0);
95  Re.max(18800.0);
96 
97  sqrSr.min(0.0016);
98  sqrSr.max(0.04);
99 
100  return 0.2*exp(- Re*sqrSr/3.6e5 - 0.12)*exp(Re*sqrSr/3.0e7);
101 }
