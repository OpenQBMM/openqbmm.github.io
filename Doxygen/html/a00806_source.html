59 void Foam::fluxFunctions::AUSMPlus::updateFluxes
60 (
61  surfaceScalarField& massFlux,
62  surfaceVectorField& momentumFlux,
63  surfaceScalarField& energyFlux,
64  const volScalarField& rho,
65  const volVectorField& U,
66  const volScalarField& H,
67  const volScalarField& p,
68  const volScalarField& a
69 )
70 {
71  surfaceVectorField normal(mesh_.Sf()/mesh_.magSf());
72  dimensionedScalar minU("smallU", dimVelocity, SMALL);
73 
74  surfaceScalarField rhoOwn
75  (
76  fvc::interpolate(rho, own_, interpScheme(rho.name()))
77  );
78  surfaceScalarField rhoNei
79  (
80  fvc::interpolate(rho, nei_, interpScheme(rho.name()))
81  );
82 
83  surfaceVectorField UOwn(fvc::interpolate(U, own_, interpScheme(U.name())));
84  surfaceVectorField UNei(fvc::interpolate(U, nei_, interpScheme(U.name())));
85 
86  surfaceScalarField HOwn(fvc::interpolate(H, own_, interpScheme(H.name())));
87  surfaceScalarField HNei(fvc::interpolate(H, nei_, interpScheme(H.name())));
88 
89  surfaceScalarField pOwn(fvc::interpolate(p, own_, interpScheme(p.name())));
90  surfaceScalarField pNei(fvc::interpolate(p, nei_, interpScheme(p.name())));
91 
92  surfaceScalarField aOwn
93  (
94  fvc::interpolate(a, own_, interpScheme(a.name()))
95  );
96  surfaceScalarField aNei
97  (
98  fvc::interpolate(a, nei_, interpScheme(a.name()))
99  );
100 
101  surfaceScalarField UvOwn(UOwn & normal);
102  surfaceScalarField UvNei(UNei & normal);
103 
104  // Compute slpit Mach numbers
105  surfaceScalarField MaOwn("MaOwn", UvOwn/max(aOwn, minU));
106  surfaceScalarField MaNei("MaNei", UvNei/max(aNei, minU));
107  surfaceScalarField magMaOwn(mag(MaOwn));
108  surfaceScalarField magMaNei(mag(MaNei));
109 
110  surfaceScalarField deltapOwn
111  (
112  "deltapOwn",
113  pos0(magMaOwn - 1)*sign(MaOwn)
114  + neg(magMaOwn - 1)
115  *MaOwn/2.0
116  *(
117  3.0
118  - sqr(MaOwn)
119  + 4.0*alpha_*sqr(sqr(MaOwn - 1.0))
120  )
121  );
122 
123  surfaceScalarField deltapNei
124  (
125  "deltapNei",
126  pos0(magMaNei - 1)*sign(MaNei)
127  + neg(magMaNei - 1)
128  *MaNei/2.0
129  *(
130  3.0
131  - sqr(MaNei)
132  + 4.0*alpha_*sqr(sqr(MaNei - 1.0))
133  )
134  );
135 
136  surfaceScalarField deltap("deltap", deltapOwn*pOwn + deltapNei*pNei);
137 
138  surfaceScalarField magMachOwn
139  (
140  "magMachOwn",
141  pos0(magMaOwn - 1)*magMaOwn
142  + neg(magMaOwn - 1)
143  *(
144  0.5*(sqr(MaOwn) + 1.0)
145  + 2.0*beta_*sqr(sqr(MaOwn) - 1.0)
146  )
147  );
148  surfaceScalarField magMachNei
149  (
150  "magMachNei",
151  pos0(magMaNei - 1)*magMaNei
152  + neg(magMaNei - 1)
153  *(
154  0.5*(sqr(MaNei) + 1.0)
155  + 2.0*beta_*sqr(sqr(MaNei) - 1.0)
156  )
157  );
158  surfaceScalarField deltaMa12
159  (
160  "deltaMa12",
161  magMachOwn - magMachNei
162  );
163  surfaceScalarField Ma12
164  (
165  "Ma12",
166  MaOwn + MaNei - deltaMa12
167  );
168 
169  surfaceScalarField a12("a12", sqrt(aOwn*aNei));
170  surfaceScalarField rhoPhi(fvc::interpolate(rho*U) & normal);
171  surfaceVectorField rhoUPhi
172  (
173  (fvc::interpolate(rho*U*U) & normal)
174  + fvc::interpolate(p)*normal
175  );
176  surfaceScalarField rhoHPhi(fvc::interpolate(rho*H*U) & normal);
177 
178  massFlux =
179  mesh_.magSf()
180  *(
181  rhoPhi
182  - 0.5*a12
183  *(
184  (0.5*deltaMa12 - mag(Ma12))*rhoOwn
185  + (0.5*deltaMa12 + mag(Ma12))*rhoNei
186  )
187  );
188 
189  momentumFlux =
190  mesh_.magSf()
191  *(
192  rhoUPhi
193  - 0.5*a12
194  *(
195  (0.5*deltaMa12 - mag(Ma12))*rhoOwn*UOwn
196  + (0.5*deltaMa12 + mag(Ma12))*rhoNei*UNei
197  )
198  );
199  - 0.5*deltap*mesh_.Sf();
200 
201  energyFlux =
202  mesh_.magSf()
203  *(
204  rhoHPhi
205  - 0.5*a12
206  *(
207  (0.5*deltaMa12 - mag(Ma12))*rhoOwn*HOwn
208  + (0.5*deltaMa12 + mag(Ma12))*rhoNei*HNei
209  )
210  );
211 }
