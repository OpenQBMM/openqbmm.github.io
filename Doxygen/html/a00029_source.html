109 void Foam::basicFieldMomentInversion::invertBoundaryMoments
110 (
111  const volScalarMomentFieldSet& moments,
112  mappedPtrList<volScalarNode>& nodes
113 )
114 {
115  // Recover reference to boundaryField of zero-order moment.
116  const volScalarField::Boundary& bf = moments[0].boundaryField();
117 
118  forAll(bf, patchi)
119  {
120  const fvPatchScalarField& m0Patch = bf[patchi];
121 
122  forAll(m0Patch, facei)
123  {
124  univariateMomentSet momentsToInvert
125  (
126  moments.size(),
127  moments.support(),
128  0.0, // Initial value
129  nFixedQuadraturePoints_
130  );
131 
132  // Copying moments from a face
133  forAll(momentsToInvert, momenti)
134  {
135  momentsToInvert[momenti]
136  = moments[momenti].boundaryField()[patchi][facei];
137  }
138 
139  // Find quadrature
140  momentInverter_().invert
141  (
142  momentsToInvert,
143  minKnownAbscissa_,
144  maxKnownAbscissa_
145  );
146 
147  label maxNodes = nodes.size();
148  label actualNodes = momentInverter_().nNodes();
149 
150  // Copy quadrature data to boundary face
151  for (label nodei = 0; nodei < maxNodes; nodei++)
152  {
153  volScalarNode& node = nodes[nodei];
154 
155  volScalarField::Boundary& weightBf
156  = node.primaryWeight().boundaryFieldRef();
157 
158  volScalarField::Boundary& abscissaBf
159  = node.primaryAbscissae()[0].boundaryFieldRef();
160 
161  if (nodei < actualNodes)
162  {
163  weightBf[patchi][facei]
164  = momentInverter_().weights()[nodei];
165 
166  abscissaBf[patchi][facei]
167  = momentInverter_().abscissae()[nodei];
168  }
169  else
170  {
171  weightBf[patchi][facei] = 0.0;
172  abscissaBf[patchi][facei] = 0.0;
173  }
174  }
175  }
176  }
177 }

179 bool Foam::basicFieldMomentInversion::invertLocalMoments
180 (
181  const volScalarMomentFieldSet& moments,
182  mappedPtrList<volScalarNode>& nodes,
183  const label celli,
184  const bool fatalErrorOnFailedRealizabilityTest
185 )
186 {
187  univariateMomentSet momentsToInvert
188  (
189  moments.size(),
190  moments.support(),
191  0.0, // Initial value
192  nFixedQuadraturePoints_
193  );
194 
195  // Copying moments from cell
196  forAll(momentsToInvert, momenti)
197  {
198  momentsToInvert[momenti] = moments[momenti][celli];
199  }
200 
201  if (!fatalErrorOnFailedRealizabilityTest)
202  {
203  if (!momentsToInvert.isRealizable(fatalErrorOnFailedRealizabilityTest))
204  {
205  return false;
206  }
207  }
208 
209  // Find quadrature
210  momentInverter_().invert
211  (
212  momentsToInvert,
213  minKnownAbscissa_,
214  maxKnownAbscissa_
215  );
216 
217  label maxNodes = nodes.size();
218  label actualNodes = momentInverter_().nNodes();
219 
220  // Recovering quadrature
221  const scalarList& weights(momentInverter_().weights());
222  const scalarList& abscissae(momentInverter_().abscissae());
223 
224  for (label nodei = 0; nodei < maxNodes; nodei++)
225  {
226  volScalarNode& node(nodes[nodei]);
227 
228  if (nodei < actualNodes)
229  {
230  node.primaryWeight()[celli] = weights[nodei];
231  node.primaryAbscissae()[0][celli] = abscissae[nodei];
232  }
233  else
234  {
235  node.primaryWeight()[celli] = 0.0;
236  node.primaryAbscissae()[0][celli] = 0.0;
237  }
238  }
239 
240  return true;
241 }
